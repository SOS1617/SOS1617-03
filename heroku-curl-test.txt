// Ver un recurso concreto

// Ver una coleccion
curl -v -XGET -H 'Content-type: application/json'  'http://sos1617-03.herokuapp.com/api/v1/results'

//Crear un elemento en una colección
curl -XPOST -H "Content-type: application/json" -d '{
"country": "Germany",
"year": "2012",
"science": "500",
"reading": "300",
"math": "400"
}' 'http://sos1617-03.herokuapp.com/api/v1/results'

//Crear un elemento en un recurso concreto (PROHIBIDO)
curl -XPOST -H "Content-type: application/json" -d '{
"country": "Germany",
"year": "2012",
"science": "500",
"reading": "300",
"math": "400"
}' 'http://sos1617-03.herokuapp.com/api/v1/results/Spain/2015'

//Actualizar una coleccion (PROHIBIDO)
curl -XPUT -H "Content-type: application/json" -d '{
	"country": "Spain",
	"year": "2012",
	"science": "1",
	"reading": "1",
	"math": "1"
}' 'http://sos1617-03.herokuapp.com/api/v1/results/'

//Actualizar un recurso concreto
curl -XPUT -H "Content-type: application/json" -d '{
	"country": "spain",
	"year": "2012",
	"science": "1",
	"reading": "1",
	"math": "1"
}' 'http://sos1617-03.herokuapp.com/api/v1/results/spain/2012'

//Borrar una coleccion
curl -v -XDELETE -H 'Content-type: application/json'  'http://sos1617-03.herokuapp.com/api/v1/results'

//Borrar un recurso concreto
curl -XDELETE -H "Content-type: application/json" 'http://sos1617-03.herokuapp.com/api/v1/results/Germany/2012'
